            with urllib.request.urlopen(req, timeout=15) as resp:

                print(f"  Chunk {i}/1: {status}")
                if i == 1:
                    data = result.get('results', [{}])[0].get('data', {})

            print(f"  Chunk {i}/1: ERROR - {e}")
        time.sleep(0.5)

    print("All chunks sent!")
