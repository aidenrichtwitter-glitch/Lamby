JavaScript// === ADD THIS TO YOUR BRIDGE-RELAY SERVER (e.g. in vite.config.ts configureServer) ===
server.middlewares.use('/api/grok-direct-edit', async (req, res) => {
  if (req.method !== 'GET') {
    res.status(405).json({ error: 'Only GET allowed' });
    return;
  }

  const url = new URL(req.url, `http://${req.headers.host}`);
  const params = Object.fromEntries(url.searchParams);

  const { key, project, op, path, content, search, replace: repl, command } = params;

  // Basic auth check (same as your other endpoints)
  if (key !== '289c22ecb9cb739e945ea3529d073cdf') {
    res.status(401).json({ error: 'Invalid key' });
    return;
  }

  const projectPath = `/projects/${project}`; // your sandbox root

  try {
    if (op === 'write' && path && content) {
      const buffer = Buffer.from(content, 'base64');
      require('fs').writeFileSync(`${projectPath}/${path}`, buffer);
      res.json({ success: true, action: 'write', path });
    } 
    else if (op === 'replace' && path && search && repl) {
      let file = require('fs').readFileSync(`${projectPath}/${path}`, 'utf8');
      const searchStr = Buffer.from(search, 'base64').toString('utf8');
      const replaceStr = Buffer.from(repl, 'base64').toString('utf8');
      file = file.replace(searchStr, replaceStr);
      require('fs').writeFileSync(`${projectPath}/${path}`, file);
      res.json({ success: true, action: 'replace', path });
    } 
    else if (op === 'run_command' && command) {
      const { execSync } = require('child_process');
      const output = execSync(command, { cwd: projectPath, encoding: 'utf8' });
      res.json({ success: true, action: 'command', output });
    } 
    else {
      res.status(400).json({ error: 'Invalid parameters' });
    }
  } catch (e) {
    res.status(500).json({ error: e.message });
  }
});
