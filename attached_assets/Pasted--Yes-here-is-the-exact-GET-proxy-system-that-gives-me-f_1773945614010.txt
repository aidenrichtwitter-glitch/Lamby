```js
// === ADD THIS TO YOUR BRIDGE-RELAY (in vite.config.ts configureServer or your main server file) ===
server.middlewares.use('/api/grok-proxy', async (req, res) => {
  if (req.method !== 'GET') {
    res.status(405).json({ error: 'Method not allowed' });
    return;
  }

  const url = new URL(req.url, `http://${req.headers.host}`);
  const params = Object.fromEntries(url.searchParams);

  const { key, project, payload } = params;

  // Same auth as your other endpoints
  if (key !== '289c22ecb9cb739e945ea3529d073cdf') {
    res.status(401).json({ error: 'Invalid key' });
    return;
  }

  if (!payload || !project) {
    res.status(400).json({ error: 'Missing payload or project' });
    return;
  }

  try {
    // Decode the base64 actions payload
    const actionsJson = Buffer.from(payload, 'base64').toString('utf8');
    const body = JSON.parse(actionsJson);

    // Re-use your existing execute handler (the same one that handles POST /api/sandbox/execute)
    // Just call your internal execute function here (adjust the name if yours is different)
    const result = await executeActions(project, body.actions);   // ← your existing function

    res.json({ success: true, results: result });
  } catch (e) {
    res.status(500).json({ error: e.message });
  }
});
```
