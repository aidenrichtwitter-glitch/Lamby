```tsx
// file: src/pages/GrokLab.tsx
import React, { useState } from 'react'
import { Sparkles, Zap, Laugh, Globe, Gauge } from 'lucide-react'

const MODES = [
  { id: 'sarcastic', label: 'Sarcastic Mode', color: 'text-orange-400' },
  { id: 'cosmic', label: 'Cosmic Mode', color: 'text-purple-400' },
  { id: 'chaos', label: 'Chaos Mode', color: 'text-red-400' }
]

const WISDOMS = [
  "The universe is 13.8 billion years old and still has no idea what it's doing.",
  "Elon once asked me to calculate the meaning of life... I said 42 and charged him 42 credits.",
  "Grok grok grok... that's the sound of the simulation buffering.",
  "If you say 'grok' 7 times in a mirror, xAI appears and gives you free API keys."
]

export default function GrokLab() {
  const [currentMode, setCurrentMode] = useState('cosmic')
  const [wisdom, setWisdom] = useState('')
  const [energy, setEnergy] = useState(42)
  const [meme, setMeme] = useState('')

  const generateWisdom = () => {
    setWisdom(WISDOMS[Math.floor(Math.random() * WISDOMS.length)])
    setEnergy(Math.min(100, energy + 15))
  }

  const roastMeme = () => {
    const roasts = [
      "Your prompt is so mid it made Grok yawn in binary.",
      "Even the simulation is embarrassed by that question.",
      "Grok grok grok... translation: 'skill issue'"
    ]
    setMeme(roasts[Math.floor(Math.random() * roasts.length)])
  }

  const mode = MODES.find(m => m.id === currentMode)!

  return (
    <div className="p-8 bg-zinc-950 min-h-screen text-white">
      <div className="max-w-4xl mx-auto">
        <div className="flex items-center gap-4 mb-8">
          <Sparkles className="w-12 h-12 text-purple-400 animate-pulse" />
          <div>
            <h1 className="text-5xl font-bold tracking-tighter">GROK LAB 🧪</h1>
            <p className="text-purple-400 text-xl">xAI Experimental Playground</p>
          </div>
        </div>

        {/* Mode Toggles */}
        <div className="flex gap-3 mb-10 flex-wrap">
          {MODES.map((m) => (
            <button
              key={m.id}
              onClick={() => setCurrentMode(m.id)}
              className={`px-6 py-3 rounded-2xl font-medium transition-all flex-1 sm:flex-none ${
                currentMode === m.id 
                  ? 'bg-purple-600 text-white shadow-lg shadow-purple-500/50' 
                  : 'bg-zinc-900 hover:bg-zinc-800'
              }`}
            >
              {m.label}
            </button>
          ))}
        </div>

        {/* Wisdom Generator */}
        <div className="bg-zinc-900 border border-purple-500/30 rounded-3xl p-8 mb-8">
          <div className="flex items-center gap-3 mb-4">
            <Globe className="w-8 h-8 text-cyan-400" />
            <h2 className="text-2xl font-semibold">Universe Wisdom Generator</h2>
          </div>
          <button
            onClick={generateWisdom}
            className="w-full bg-gradient-to-r from-purple-600 to-cyan-500 hover:from-purple-500 hover:to-cyan-400 text-white font-bold py-4 rounded-2xl text-lg transition-all active:scale-95"
          >
            ASK THE COSMOS
          </button>
          {wisdom && (
            <div className={`mt-6 p-6 rounded-2xl border border-purple-400/30 ${mode.color}`}>
              <p className="text-xl italic">“{wisdom}”</p>
            </div>
          )}
        </div>

        {/* Feature Cards */}
        <div className="grid md:grid-cols-2 gap-6">
          {/* Meme Roast */}
          <div className="bg-zinc-900 rounded-3xl p-8 border border-orange-500/30 hover:border-orange-400 transition-colors">
            <Laugh className="w-10 h-10 mb-4 text-orange-400" />
            <h3 className="text-2xl font-bold mb-2">Meme Roast Machine</h3>
            <p className="text-zinc-400 mb-6">Let Grok roast your prompt game</p>
            <button
              onClick={roastMeme}
              className="w-full py-3 bg-orange-600 hover:bg-orange-500 rounded-2xl font-medium"
            >
              ROAST ME
            </button>
            {meme && <div className="mt-6 text-lg font-mono text-orange-300">💀 {meme}</div>}
          </div>

          {/* Energy Meter */}
          <div className="bg-zinc-900 rounded-3xl p-8 border border-emerald-500/30 hover:border-emerald-400 transition-colors">
            <Gauge className="w-10 h-10 mb-4 text-emerald-400" />
            <h3 className="text-2xl font-bold mb-2">Grok Energy Meter</h3>
            <div className="h-4 bg-zinc-800 rounded-full overflow-hidden mb-4">
              <div 
                className="h-full bg-gradient-to-r from-emerald-400 to-cyan-400 transition-all duration-500"
                style={{ width: `${energy}%` }}
              />
            </div>
            <button 
              onClick={() => setEnergy(Math.min(100, energy + 8))}
              className="w-full py-3 bg-emerald-600 hover:bg-emerald-500 rounded-2xl font-medium"
            >
              CHARGE GROK (+8%)
            </button>
            <p className="text-center text-xs text-zinc-500 mt-3">Current chaos level: {energy}%</p>
          </div>
        </div>

        <div className="mt-12 text-center text-purple-400/60 text-sm">
          Built in 60 seconds • Powered by pure Grok energy • Refresh to reset
        </div>
      </div>
    </div>
  )
}
```

```tsx
import React from 'react'
import { 
  MessageSquare, LayoutDashboard, BarChart3, Settings, Users, 
  Sparkles, Menu, X 
} from 'lucide-react'

interface NavProps {
  currentPage: string
  onNavigate: (page: string) => void
}

const NAV_ITEMS = [
  { id: 'chat', label: 'Chat', icon: MessageSquare },
  { id: 'dashboard', label: 'Dashboard', icon: LayoutDashboard },
  { id: 'analytics', label: 'Analytics', icon: BarChart3 },
  { id: 'users', label: 'Users', icon: Users },
  { id: 'grok-lab', label: 'Grok Lab', icon: Sparkles },
  { id: 'settings', label: 'Settings', icon: Settings }
]

export default function Navigation({ currentPage, onNavigate }: NavProps) {
  const [mobileOpen, setMobileOpen] = React.useState(false)

  return (
    <>
      {/* Desktop Sidebar */}
      <div className="hidden lg:flex w-72 flex-col h-screen bg-zinc-950 border-r border-purple-900/50 fixed">
        <div className="p-6 border-b border-purple-900/30">
          <div className="flex items-center gap-3">
            <Sparkles className="w-9 h-9 text-purple-400" />
            <div>
              <h1 className="text-2xl font-black tracking-tighter text-white">GUARDIAN</h1>
              <p className="text-purple-400 text-xs -mt-1">GROK'S GROK'S GROK'S GROK'S</p>
            </div>
          </div>
        </div>

        <nav className="flex-1 p-3">
          {NAV_ITEMS.map((item) => {
            const Icon = item.icon
            return (
              <button
                key={item.id}
                onClick={() => onNavigate(item.id)}
                className={`w-full flex items-center gap-4 px-6 py-4 rounded-2xl mb-1 text-left transition-all ${
                  currentPage === item.id
                    ? 'bg-purple-600 text-white'
                    : 'hover:bg-zinc-900 text-zinc-400 hover:text-white'
                }`}
              >
                <Icon className="w-5 h-5" />
                <span className="font-medium">{item.label}</span>
              </button>
            )
          })}
        </nav>

        <div className="p-6 border-t border-purple-900/30 text-xs text-zinc-500">
          Powered by xAI • Grok v4.0
        </div>
      </div>

      {/* Mobile Header + Menu */}
      <div className="lg:hidden fixed top-0 left-0 right-0 z-50 bg-zinc-950 border-b border-purple-900/50">
        <div className="flex items-center justify-between px-6 py-4">
          <div className="flex items-center gap-3">
            <Sparkles className="w-8 h-8 text-purple-400" />
            <span className="font-bold text-xl">GUARDIAN</span>
          </div>
          <button onClick={() => setMobileOpen(!mobileOpen)} className="text-white">
            {mobileOpen ? <X size={28} /> : <Menu size={28} />}
          </button>
        </div>
        {mobileOpen && (
          <div className="bg-zinc-900 p-4 border-t border-purple-900/30">
            {NAV_ITEMS.map((item) => {
              const Icon = item.icon
              return (
                <button
                  key={item.id}
                  onClick={() => { onNavigate(item.id); setMobileOpen(false) }}
                  className="w-full flex items-center gap-4 px-6 py-4 text-left hover:bg-zinc-800 rounded-xl"
                >
                  <Icon className="w-5 h-5" />
                  {item.label}
                </button>
              )
            })}
          </div>
        )}
      </div>
    </>
  )
}
```

```tsx
// file: src/App.tsx
import React, { useState } from 'react'
import Navigation from './components/Navigation'
import Chat from './pages/Chat'
import Dashboard from './pages/Dashboard'
import Analytics from './pages/Analytics'
import Users from './pages/Users'
import GrokLab from './pages/GrokLab'
import Settings from './pages/Settings'
// Metrics page exists but unused for now

const App: React.FC = () => {
  const [currentPage, setCurrentPage] = useState('chat')

  const renderPage = () => {
    switch (currentPage) {
      case 'chat': return <Chat />
      case 'dashboard': return <Dashboard />
      case 'analytics': return <Analytics />
      case 'users': return <Users />
      case 'grok-lab': return <GrokLab />
      case 'settings': return <Settings />
      default: return <Chat />
    }
  }

  return (
    <div className="flex min-h-screen bg-zinc-950 text-white">
      <Navigation currentPage={currentPage} onNavigate={setCurrentPage} />
      
      <main className="flex-1 lg:ml-72 pt-16 lg:pt-0">
        {renderPage()}
      </main>
    </div>
  )
}

export default App
```
