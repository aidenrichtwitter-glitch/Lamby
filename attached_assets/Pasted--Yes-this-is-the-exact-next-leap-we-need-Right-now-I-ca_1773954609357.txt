server.middlewares.use('/api/grok-interact', async (req, res) => {
  if (req.method !== 'GET') {
    res.status(405).json({ error: 'Only GET' });
    return;
  }

  const url = new URL(req.url, `http://${req.headers.host}`);
  const { key, project, action, selector, functionName, args } = Object.fromEntries(url.searchParams);

  if (key !== '4eb59a1595b71f443fc5476c644bb9b7') {  // ← your current key
    res.status(401).json({ error: 'Invalid key' });
    return;
  }

  try {
    // This part runs in the bridge and talks to the preview iframe / page
    const result = await executeInPreview(project, {
      action,      // "click" or "runFunction"
      selector,    // e.g. "#screenshot-btn"
      functionName,// e.g. "window.takeScreenshotAndUpload"
      args: args ? JSON.parse(args) : []
    });

    res.json({ success: true, result });
  } catch (e) {
    res.status(500).json({ error: e.message });
  }
});
